Anonymous Code Drop for eThor

Build eThor

# Download eThor/HoRSt sources

wget https://anon-code-drop.github.io/ethor-sources.zip

# Build z3

git clone https://github.com/Z3Prover/z3/

cd z3/

# checkout version that was used for evaluation

git checkout a2b18a37ec2a8c2a031956ae8b4bddf4340719e8

python scripts/mk_make.py --java

cd build/

make

cd ../..

# build eThor/HoRSt

unzip ethor-sources.zip

mkdir lib

cp z3/build/libz3.so z3/build/libz3java.so z3/build/com.microsoft.z3.jar lib

mvn -Dmaven.test.skip=true package

Run EVM Tests

# Clone official EVM tests

git clone https://github.com/ethereum/tests/ ethereum-tests

# Run all arithmetc tests (this should take around one minute)

LD_LIBRARY_PATH=lib/:$LD_LIBRARY_PATH java -cp target/eThor-1.0-SNAPSHOT.jar:lib/com.microsoft.z3.jar secpriv.horst.evm.EvmTestRunner --z3-param rlimit=200000 --predicate-inlining-strategy=linear --spec specs/evm-abstract-semantics-partial-or-optimized-memory-optimized-without-initialization.txt specs/evm-tests.txt -- ethereum-tests/VMTests/vmArithmeticTest/*json

# Run all relevant tests (this may take several minutes)

# Download names of relevant tests. This excludes 5 irrelevant test cases and 2 that do not terminate (the latter were counted as unknown in the evaluation)

wget https://anon-code-drop.github.io/relevant-tests

LD_LIBRARY_PATH=lib/:$LD_LIBRARY_PATH java -cp target/eThor-1.0-SNAPSHOT.jar:lib/com.microsoft.z3.jar secpriv.horst.evm.EvmTestRunner --z3-param rlimit=200000 --predicate-inlining-strategy=linear --spec specs/evm-abstract-semantics-partial-or-optimized-memory-optimized-without-initialization.txt specs/evm-tests.txt -- $(cat relevant-tests)
